     private String reportDescription;
 
     private String username;
 
     private String title;
 
     @ManyToOne(cascade = CascadeType.ALL)
     @JoinColumn(name = "report_id")
     @JsonIgnore
     private Post reportPost;

         this.title = title;
         this.reportDescription = description;
     }

 }
